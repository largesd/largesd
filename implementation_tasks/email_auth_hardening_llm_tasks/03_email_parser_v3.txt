Goal:
Extend the email submission parser to support BDA Submission v3, including
Auth-Token unwrapping, Payload-Hash extraction, and a processor-facing
parse_for_processor wrapper that returns structured decisions instead of raw
exceptions.

Why this matters:
The processor needs to distinguish unparseable spam from legacy v1/v2, from
malformed v3, from valid v3. Each category has different ack behavior. The
parser must also safely unwrap multi-line JWTs that mail clients may wrap.

4. Implement parse_for_processor
   Add to EmailSubmissionParser:

     def parse_for_processor(
         self,
         body: str,
         submitter_email: str,
         subject: str,
         require_auth: bool = True,
         allow_legacy: bool = False,
     ) -> EmailSubmissionParseResult:
         lines = body.replace("\r\n", "\n").split("\n")
         if not lines:
             return EmailSubmissionParseResult(
                 decision="drop", version_hint="unknown",
                 reason_code="unrecognized", ack_safe=False
             )

         stripped_first = lines[0].strip()
         if self.HEADER_RE.match(stripped_first):
             version = "v1"
         elif self.HEADER_V2_RE.match(stripped_first):
             version = "v2"
         elif self.HEADER_V3_RE.match(stripped_first):
             version = "v3"
         else:
             return EmailSubmissionParseResult(
                 decision="drop", version_hint="unknown",
                 reason_code="unrecognized", ack_safe=False
             )

         if version in ("v1", "v2"):
             if require_auth and not allow_legacy:
                 return EmailSubmissionParseResult(
                     decision="reject", version_hint=version,
                     reason_code="legacy_disabled", ack_safe=False
                 )
             try:
                 submission = self.parse_body(body, submitter_email, subject)
                 return EmailSubmissionParseResult(
                     decision="accept", version_hint=version,
                     submission=submission, ack_safe=True
                 )
             except EmailSubmissionError as exc:
                 return EmailSubmissionParseResult(
                     decision="reject", version_hint=version,
                     reason_code="invalid_body", ack_safe=False,
                     exception=exc
                 )

         # v3 path
         try:
             submission = self._parse_v3(lines, body, submitter_email, subject)
         except EmailSubmissionError as exc:
             # Determine if missing token => missing_token, else invalid_body
             reason = "invalid_body"
             if "Auth-Token" in str(exc) or "Missing required fields" in str(exc):
                 missing = self.REQUIRED_FIELDS_V3 - self._extract_v3_headers(lines)
                 if "Auth-Token" in missing:
                     reason = "missing_token"
                 elif missing:
                     reason = "missing_required_field"
             return EmailSubmissionParseResult(
                 decision="reject", version_hint="v3",
                 reason_code=reason, ack_safe=False, exception=exc
             )

         if require_auth and not submission.auth_token:
             return EmailSubmissionParseResult(
                 decision="reject", version_hint="v3",
                 reason_code="missing_token", ack_safe=False
             )

         return EmailSubmissionParseResult(
             decision="accept", version_hint="v3",
             submission=submission, ack_safe=True
         )

5. Implement _parse_v3
   This is similar to _parse_v1 but uses REQUIRED_FIELDS_V3 and extracts the
   new fields. It must also handle Auth-Token line wrapping:

     def _parse_v3(self, lines, body, submitter_email, subject):
         headers: dict[str, str] = {}
         i = 1
         auth_token_parts: list[str] = []
         in_auth_token = False

         while i < len(lines):
             line = lines[i]
             if in_auth_token:
                 if line.startswith(" ") or line.startswith("\t"):
                     auth_token_parts.append(line.lstrip(" \t"))
                     i += 1
                     continue
                 else:
                     # End of Auth-Token continuation
                     if auth_token_parts:
                         headers["Auth-Token"] = "".join(auth_token_parts)
                     in_auth_token = False
                     # Fall through to normal field parsing for this line
             m = self.FIELD_RE.match(line)
             if m:
                 key, value = m.group(1), m.group(2).strip()
                 if key == "Auth-Token":
                     in_auth_token = True
                     auth_token_parts = [value]
                 else:
                     headers[key] = value
                 i += 1
             elif line.strip() == "":
                 if in_auth_token:
                     if auth_token_parts:
                         headers["Auth-Token"] = "".join(auth_token_parts)
                     in_auth_token = False
                 i += 1
                 break
             else:
                 if in_auth_token:
                     if auth_token_parts:
                         headers["Auth-Token"] = "".join(auth_token_parts)
                     in_auth_token = False
                 i += 1
                 break

         if in_auth_token and auth_token_parts:
             headers["Auth-Token"] = "".join(auth_token_parts)

         missing = self.REQUIRED_FIELDS_V3 - set(headers.keys())
         if missing:
             raise EmailSubmissionError(f"Missing required fields: {', '.join(missing)}")

         side = headers["Position"].upper().strip()
         if side not in self.VALID_SIDES:
             raise EmailSubmissionError(f"Position must be FOR or AGAINST, got: {headers['Position']}")

         remaining = "\n".join(lines[i:])
         facts = self._extract_section(remaining, "Facts")
         inference = self._extract_section(remaining, "Inference")
         counter_arguments = self._extract_section(remaining, "Counter-Arguments")

         if not facts or not facts.strip():
             raise EmailSubmissionError("Facts section is required and cannot be empty.")
         if not inference or not inference.strip():
             raise EmailSubmissionError("Inference section is required and cannot be empty.")

         topic_id = headers["Topic-Area"].strip().lower()
         if not re.match(r"^[a-z0-9_-]+$", topic_id):
             raise EmailSubmissionError(f"Invalid Topic-Area format: {topic_id}")

         # Strip all internal whitespace from auth token before returning
         raw_token = headers.get("Auth-Token", "")
         cleaned_token = "".join(raw_token.split()) if raw_token else None

         return EmailSubmission(
             debate_id=headers["Debate-ID"].strip(),
             resolution=headers["Resolution"].strip(),
             submission_id=headers["Submission-ID"].strip(),
             submitted_at=headers["Submitted-At"].strip(),
             side=side,
             topic_id=topic_id,
             facts=facts.strip(),
             inference=inference.strip(),
             counter_arguments=counter_arguments.strip() if counter_arguments else None,
             submitter_email=submitter_email,
             subject=subject,
             auth_token=cleaned_token,
             payload_hash=headers.get("Payload-Hash"),
             payload_hash_alg=headers.get("Payload-Hash-Alg"),
         )

Acceptance criteria:
- Parser correctly identifies v3 header.
- Parser extracts Auth-Token, Payload-Hash, Payload-Hash-Alg from v3.
- Parser rejects v3 missing Auth-Token with reason_code="missing_token".
- Parser rejects v3 missing other required fields with
  reason_code="missing_required_field".
- Parser correctly unwraps a multi-line Auth-Token wrapped by a mail client.
- Parser rejects a malformed wrapped token.
- parse_for_processor returns decision="drop" for unknown mail.
- parse_for_processor returns decision="reject" for legacy v1/v2 when
  require_auth=True and allow_legacy=False.
- parse_for_processor returns decision="accept" for valid v3.
- Existing v1/v2 parser tests still pass.
- build_email_body v1 behavior is unchanged for existing callers.

Guardrails:

- Do not break existing _extract_section behavior.
- When unwrapping Auth-Token, strip ALL whitespace (spaces, tabs, newlines)
  from the concatenated value before returning.
